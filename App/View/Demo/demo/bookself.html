模块代码

  1. define(function(require, exports, module){
  3.     //seajs加载脚本
  4.     require("jquery.min");
  5.     require("underscore-min");
  6.     require("backbone-min");
  8.     //模板
  9.     var temp = '<li class="item">书名：<%=name%>价格：<%=price%>出版社：<%=publisher%><a href="javascript:void(0)" class="btn btn-danger" data-name="<%=name%>">删除</a></li>';
  11.     //图书模型
  12.     var Book  = Backbone.Model.extend({
  13.         name:"",
  14.         price:0,
  15.         publisher:""
  16.     });
  17.     var book = new Book();
  19.     //图书类
  20.     var BookSelf = Backbone.Collection.extend({
  21.         Model:Book
  22.     });
  24.     var bookSelf = new BookSelf();
  26.     //异步加载数据
  27.     bookSelf.url = "/resource/data/books.json";
  28.     bookSelf.fetch({
  29.         success:function(collection,response,options){
  30.             collection.each(function(b){
  31.                 book.set({name:b.get('name'),price:b.get('price'),publisher:b.get('publisher')});
  32.             });
  33.         },
  34.         error:function(collection,response,options){
  35.             alert("error");
  36.         }
  37.     });
  39.     //对应着每一条数据
  40.     var ItemView = Backbone.View.extend({
  41.         el:$("#book-list"),
  42.         initialize:function(){
  43.             this.listenTo(this.model, "change", this.addOne);
  44.         },
  45.         events:{
  46.             "click .btn-danger" : "delete"
  47.         },
  48.         delete:function(e){
  49.             $(e.target).parent("li").remove();
  50.         },
  51.         addOne:function(){
  52.             this.render(this.model.toJSON());
  53.         },
  54.         render:function(obj){
  55.             $("#book-list").append(_.template(temp)(obj));
  56.         }
  57.     });
  59.     /*显示整体*/
  60.     var AppView = Backbone.View.extend({
  61.         el:$(".container"),
  62.         //new 一个view时会自动加载这个方法
  63.         initialize:function(){
  64.             var itemView = new ItemView({model:book});
  65.         },
  66.         events:{
  67.             "click #add" :  "add"
  68.         },
  69.         add:function(){
  70.             var name = $("#name").val();
  71.             var price = $("#price").val();
  72.             var publisher = $("#publisher").val();
  73.             book.set({name:name,price:price,publisher:publisher});
  74.             bookSelf.add(book);
  75.             $("#name,#price,#publisher").val("");
  76.         }
  77.     });
  79.     var app = new AppView();
  81. });
